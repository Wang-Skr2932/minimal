# -*- coding: utf-8 -*- # โปรแกรมคำนวณปริมาณยาน้ำสำหรับเด็ก def calculate_dose_mg(weight_kg, dose_per_kg_mg): return weight_kg * dose_per_kg_mg def calculate_volume_ml(dose_mg, concentration_mg_per_ml): return dose_mg / concentration_mg_per_ml def get_drug_info(choice): drug_data = { "1": { "name": "Paracetamol", "dose_per_kg": 15, # 10–15 mg/kg/ครั้ง "concentration": 24 # 120mg/5ml = 24mg/ml }, "2": { "name": "CPM (Chlorpheniramine)", "dose_per_kg": 0.35, # mg/kg/วัน แบ่ง 3 ครั้ง "concentration": 0.4 # 2mg/5ml = 0.4mg/ml }, "3": { "name": "Azithromycin", "dose_per_kg": 10, # 10 mg/kg/วัน "concentration": 40 # 200mg/5ml = 40mg/ml }, "4": { "name": "Carbocysteine", "dose_per_kg": 10, # 10 mg/kg/ครั้ง "concentration": 50 # 250mg/5ml = 50mg/ml }, "5": { "name": "Guaifenesin (Glyceryl Guaiacolate)", "dose_per_kg": 6.6, # ประมาณ 6.6 mg/kg/ครั้ง "concentration": 20 # 100mg/5ml = 20mg/ml } } return drug_data.get(choice) # เริ่มโปรแกรม print("=== โปรแกรมคำนวณยาน้ำสำหรับเด็ก ===") print("เลือกยาที่ต้องการคำนวณ:") print("1. Paracetamol") print("2. CPM (Chlorpheniramine)") print("3. Azithromycin") print("4. Carbocysteine") print("5. Guaifenesin (Glyceryl Guaiacolate)") choice = input("กรุณาเลือกหมายเลขยา (1-5): ") drug = get_drug_info(choice) if not drug: print("เลือกไม่ถูกต้อง") else: try: weight = float(input("ใส่น้ำหนักเด็ก (กิโลกรัม): ")) dose_mg = calculate_dose_mg(weight, drug["dose_per_kg"]) volume_ml = calculate_volume_ml(dose_mg, drug["concentration"]) print("\nยาที่เลือก: {}".format(drug["name"])) print("ขนาดยา: {:.2f} mg".format(dose_mg)) print("ปริมาณที่ควรให้: {:.2f} ml".format(volume_ml)) except ValueError: print("กรุณากรอกน้ำหนักเป็นตัวเลข")
